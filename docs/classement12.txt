function calcule_nbre_class(&$tab, $joueur, $adversaire)
{
	   $posjou=0;
	   $posadv=0;
      
	   for($i=0;$i < count($tab) ;$i++) 
	   { 

	        if( $tab[$i] == $joueur) {

	            $posjou=$i;

	            //echo'Position du joueur dans le tableau'+$posjou;

	        }elseif ($tab[$i] == $adversaire) {

	            $posadv=$i;

	            //echo'Position de l\' adversaire dans le tableau'+$posadv;

	  
	        }

	     }

	    return   $posjou - $posadv;
}
